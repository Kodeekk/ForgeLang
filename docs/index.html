Quick Example

module hello;

import [println] from std.io;
import std.math;

// Define an enum with associated data
enum Shape {
    Circle(radius: f64),
    Rect(width: f64, height: f64)
}

// Pattern matching
fn area(s: Shape) -> f64 {
    return match s {
        Shape.Circle(r) => math.PI * r * r,
        Shape.Rect(w, h) => w * h
    };
}

fn main() -> int {
    var shapes = [
        Shape.Circle(5.0),
        Shape.Rect(4.0, 6.0)
    ];
    
    for shape in shapes {
        println("Area: {area(shape)}");
    }
    
    return 0;
}
